Create a modern, minimalistic streetwear e-commerce web app for a brand called "XF".

Use Supabase as backend.

---

DATABASE STRUCTURE (IMPORTANT):

Use the following SQL schema:

-- UUID Extension
create extension if not exists "pgcrypto";

create table users (
id uuid primary key default gen_random_uuid(),
name text,
email text unique,
password text,
role text default 'user',
shipping_address text,
created_at timestamp default now()
);

create table admins (
id uuid primary key default gen_random_uuid(),
name text,
email text unique,
role text default 'worker',
permissions jsonb default '{}'::jsonb
);

create table products (
id uuid primary key default gen_random_uuid(),
name text,
price numeric,
category text,
description text,
image_url text,
stock integer default 0
);

create table orders (
id uuid primary key default gen_random_uuid(),
user_id uuid references users(id),
total_price numeric,
status text default 'pending',
shipping_address text,
created_at timestamp default now()
);

create table order_items (
id uuid primary key default gen_random_uuid(),
order_id uuid references orders(id),
product_id uuid references products(id),
quantity integer,
price numeric
);

create table tickets (
id uuid primary key default gen_random_uuid(),
user_id uuid references users(id),
subject text,
status text default 'open',
created_at timestamp default now()
);

create table ticket_messages (
id uuid primary key default gen_random_uuid(),
ticket_id uuid references tickets(id),
sender_role text,
message text,
created_at timestamp default now()
);

Disable Row Level Security (RLS) for development.
